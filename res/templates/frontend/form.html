    // Show inline messages for each field
    ###MSG_###


    // Add error class to each field
    ###ERR_###

    ###VALUE_###


    // Set state of a checkbox
    ###CHECKED_###

    ###CHECKED__[######]###

    ###SELECTED__[######]###

    ###LABEL_###
